def zip2coor(x):
    loca = geolocator.geocode('CA:'+str(x),timeout=20) 
    # as we are in California added CA:. otherwise work not in all cases
    return Series({'lat':loca.latitude,'long':loca.longitude, 'zip':int(x)})

zipcoor = zcit.apply(zip2coor)
