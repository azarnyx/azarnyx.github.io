def draw_but(but1, change1):
    if change1:
        if but1.get_state()['description'] == 'Zoom in!':
            but1.close()
            button = widgets.Button(description = "Zoom out!")
            display(button)
            button.on_click(Zoomout)
        elif but1.get_state()['description'] == 'Zoom out!':
            but1.close()
            button = widgets.Button(description = "Zoom in!")
            display(button)
            button.on_click(Zoomin)
    
def Zoomout(but):
    clear_output()
    fig, ax = plt.subplots(figsize=(16, 12))
